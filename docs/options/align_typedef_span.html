align_typedef_span

Properties

  Category: Code alignment options (not left column spaces/tabs)
      Type: numeric [0 .. 16]
   Default: 0

Description

The span for aligning single-line typedefs.

0: Don't align (default).

Examples

raw CPP code align_typedef_span=0 align_typedef_span=1 align_typedef_span=2 align_typedef_span=3
typedef char*t; typedef unsigned char*t; //1 typedef int*t; typedef unsigned*t; //1 //2 typedef float*t; typedef long double*t; //1 //2 //3 typedef long long double*t; typedef double double*t; typedef char*t; typedef unsigned char*t; //1 typedef int*t; typedef unsigned*t; //1 //2 typedef float*t; typedef long double*t; //1 //2 //3 typedef long long double*t; typedef double double*t; typedef char* t; typedef unsigned char*t; //1 typedef int* t; typedef unsigned*t; //1 //2 typedef float* t; typedef long double*t; //1 //2 //3 typedef long long double*t; typedef double double* t; typedef char* t; typedef unsigned char*t; //1 typedef int* t; typedef unsigned* t; //1 //2 typedef float* t; typedef long double*t; //1 //2 //3 typedef long long double*t; typedef double double* t; typedef char* t; typedef unsigned char*t; //1 typedef int* t; typedef unsigned* t; //1 //2 typedef float* t; typedef long double* t; //1 //2 //3 typedef long long double*t; typedef double double* t;
